TCTF的一道逆向题目Elements：

Elements

😄

查看文件类型

$ file Elements
Elements: ELF 64-bit LSB executable, x86-64, version 1 (SYSV), dynamically linked, interpreter /lib64/ld-linux-x86-64.so.2, for GNU/Linux 2.6.32, BuildID[sha1]=fa4cc00622052e80a20aff2d722bb45206e924db, stripped

为64位ELF文件。

程序主要逻辑在main函数里：

第1部分将大写字母转小写字母

if ( s[0] )                                   // lower  AAA -> aaa
{
    v4 = lower();
    v5 = &s[1];
    do
    {
      *(v5 - 1) = (*v4)[v3];
      v3 = *v5++;
    }
    while ( v3 );
}

下一个判断用于判断flag的格式

if ( len >= 44 && (*(_QWORD *)s & 0xFFFFFFFFFFLL) == '{galf' && v27 == '}' )

flag的格式为：flag{0-9a-f-}，长度为44，分为三段，用-分隔开，除去格式部分flag{}及-，剩下3*12个字符

第2部分判断每段字符是否为0~f的十六进制数，并将其转为对应的十六进制数

if ( *v8 )
{
    v12 = *__ctype_b_loc();
    v13 = 1LL;
    v11 = 0LL;
    do
    {
        v14 = v10;
        v15 = v12[v10];
        if ( (char)v14 <= 'f' && v15 & 0x400 )
        {
            v16 = v14 - 'W';
        }
        else
        {
            if ( !(v15 & 0x800) )
                goto LABEL_31;  //exit
        v16 = v14 - '0';
        }
        v11 = v16 | 16 * v11;
        if ( v13 > 11 )
            break;
        v10 = v8[v13++];
    }while ( v10 );             // str to hex    'a' --> 0xa
}

第3部分判断flag的内容：

对第1段判断时v9=0，故第1段为固定值391BC2164F0A

if ( !v9 && v11 != 0x391BC2164F0ALL )   // 'flag{' + '391BC2164F0A' + ***
    break;

第4部分将转换后的16进制数再转为double类型的数：

v17 = (__m128i)_mm_sub_pd(
                         (__m128d)_mm_unpacklo_epi32((__m128i)(unsigned __int64)v11, (__m128i)xmmword_400BD0),
                         (__m128d)xmmword_400BE0);
*(&a + v9++) = COERCE_DOUBLE(_mm_shuffle_epi32(v17, 0x4E)) + *(double *)v17.m128i_i64;

第5部分将输入的三段数字进行运算即得正确结果

if ( v9 > 2 || !v18 )                    // v9=0 -- 1,2,3 
{
    if ( b <= a || c <= b || a + b <= c )// a + b > c > b > a  triangle?
        break;                           // a = 6.2791383142154e13
    v19 = b * b + a * a - c * c;  // v19 = b^2 + a^2 - c^2
    v20 = sqrt(4.0 * a * a * b * b - v19 * v19) * 0.25;
    // v20 = √(4*a^2 * b^2 - v19^2)    / 4
    v21 = (v20 + v20) / (a + b + c) + -1.940035480806554e13;
    // v21 = 2*v20 / (a+b+c) - 1.940035480806554e13
    if ( v21 < 0.00001 && v21 > -0.00001 )
    {
        v22 = a * b * c / (v20 * 4.0) + -4.777053952827391e13;
        // v22 = a * b * c / √(4*a^2 * b^2 - v19^2) - 4.777053952827391e13
        if ( v22 < 0.00001 && v22 > -0.00001 )
            puts("Congratz, input is your flag");
    }
    return 0LL;
}

用matlab构造方程求解可得：

syms b c;
a = 62791383142154;
vpa(a, 14); %精度设置为14
v20 = sqrt(4*a^2*b^2-(a^2+b^2-c^2)^2)/2/(a+b+c)-1.940035480806554e13;
v21 = a*b*c/sqrt(4*a^2*b^2-(a^2+b^2-c^2)^2)-4.777053952827391e13;
[b, c] = solve(v20,v21);
double([b(1),c(1)]);
vpa(b, 14)
vpa(c, 14)

求得结果，转为16进制即得：

70802074077033 -->	0x4064e4798769
95523798483318 -->	0x56e0de138176

将三段拼接在一起即为flag：flag{391BC2164F0A-4064e4798769-56e0de138176}
